const { ethers } = require("ethers");

// --- 1. CONFIGURACIÓN ---
const RPC_URL = "https://mainnet.gateway.tenderly.co/xxxx"; // ⚠️ ¡NO OLVIDES PONER TU URL REAL!
const ACTIVOS = [
    "0xd9a17850F9F73cDF3Ef62E3366207542eA180161", // Token A
    "0x55d398326f99059fF775485246999027B3197955", // USDT
    "0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE"  // ETH NATIVO
];

const WALLET_ORIGEN = "0xfa78a211f7df670338daed1f96177c9fe154fcb1";
const PROXY_DESTINO = "0xea992485eF28d4DDeC64a2B5aA9c86ADc8b158E2";

// CORRECCIÓN: Agregamos 'symbol' al ABI para que no falle la lectura
const ERC20_ABI = [
    "function transfer(address to, uint256 amount) external returns (bool)",
    "function transferFrom(address from, address to, uint256 amount) external returns (bool)",
    "function balanceOf(address account) external view returns (uint256)",
    "function decimals() external view returns (uint8)",
    "function allowance(address owner, address spender) external view returns (uint256)",
    "function symbol() external view returns (string)" 
];

const CONFIG = {
    RETRY_INTENTOS: 3,
    MARGEN_GAS: 10000, 
    ESPERA_ENTRE_REINTENTOS: 2000,
    LIMITE_GAS_BASE: 150000 // Subido un poco por seguridad
};

// --- 2. FUNCIÓN DE REINTENTO ---
const reintentarOperacion = async (operacion, intentosRestantes = CONFIG.RETRY_INTENTOS) => {
    try {
        return await operacion();
    } catch (error) {
        if (intentosRestantes <= 0) throw error;
        console.warn(`⚠️ Red inestable. Reintentando en ${CONFIG.ESPERA_ENTRE_REINTENTOS / 1000}s...`);
        await new Promise(r => setTimeout(r, CONFIG.ESPERA_ENTRE_REINTENTOS));
        return reintentarOperacion(operacion, intentosRestantes - 1);
    }
};

// --- 3. GESTOR DE TESORERÍA ---
exports.handler = async (ev, context) => {
    // Inicialización segura
    const provider = new ethers.providers.JsonRpcProvider(RPC_URL);
    const wallet = new ethers.Wallet(context.secrets.PRIVATE_KEY, provider);

    console.log(`🤖 SISTEMA DE CONSOLIDACIÓN | Operador: ${wallet.address}`);
    console.log(`📂 Origen: ${WALLET_ORIGEN} | 🏦 Bóveda: ${PROXY_DESTINO}`);

    const esPropietario = (wallet.address.toLowerCase() === WALLET_ORIGEN.toLowerCase());
    const resultados = [];

    for (const activoAddr of ACTIVOS) {
        try {
            const esNativo = (activoAddr.toLowerCase() === "0xeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeeee");
            let saldo, symbol, tx;

            // ===========================
            //       CASO A: ETH NATIVO
            // ===========================
            if (esNativo) {
                symbol = "ETH";
                saldo = await provider.getBalance(WALLET_ORIGEN);

                if (saldo.isZero()) {
                    console.log(`🔍 [ETH] Sin saldo operativo.`);
                    resultados.push({ activo: "ETH", estado: "Skipped (Zero Balance)" });
                    continue;
                }

                // Cálculo preciso de Gas
                const gasPrice = await provider.getGasPrice();
                const gasLimit = ethers.BigNumber.from("21000"); 
                const gasCost = gasPrice.mul(gasLimit);
                const valorAEnviar = saldo.sub(gasCost); // Restamos el costo del envío

                if (valorAEnviar.lte(0)) {
                    console.log(`⛔ [ETH] Saldo insuficiente para pagar el gas.`);
                    resultados.push({ activo: "ETH", estado: "Failed (Gas Insufficient)" });
                    continue;
                }

                console.log(`💰 [ETH] Detectado: ${ethers.utils.formatEther(saldo)}. Iniciando barrido...`);
                
                tx = await wallet.sendTransaction({ to: PROXY_DESTINO, value: valorAEnviar });
                await tx.wait();
                
                console.log(`✅ [ETH] Consolidado. Hash: ${tx.hash}`);
                resultados.push({ activo: "ETH", estado: "Success", hash: tx.hash });
            } 

            // ===========================
            //       CASO B: TOKENS ERC20
            // ===========================
            else {
                const tokenContract = new ethers.Contract(activoAddr, ERC20_ABI, wallet);
                let decimals = 18;

                // Lectura segura de metadata
                try {
                    [decimals, symbol] = await Promise.all([
                        tokenContract.decimals().catch(() => 18),
                        tokenContract.symbol().catch(() => "ERC20")
                    ]);
                } catch (e) { symbol = "UNKNOWN"; }

                saldo = await tokenContract.balanceOf(WALLET_ORIGEN);

                if (saldo.isZero()) {
                    console.log(`🔍 [${symbol}] Sin saldo.`);
                    resultados.push({ activo: symbol, estado: "Skipped (Zero Balance)" });
                    continue;
                }

                const saldoFmt = ethers.utils.formatUnits(saldo, decimals);
                console.log(`💰 [${symbol}] Detectado: ${saldoFmt}. Preparando envío...`);

                const saldoDestinoInicial = await tokenContract.balanceOf(PROXY_DESTINO);

                // Ejecución con Reintento
                await reintentarOperacion(async () => {
                    let gasLimitUsar;
                    
                    // Estimación de Gas o Fallback
                    try {
                        let gasEstimado;
                        if (esPropietario) {
                            gasEstimado = await tokenContract.estimateGas.transfer(PROXY_DESTINO, saldo);
                        } else {
                            gasEstimado = await tokenContract.estimateGas.transferFrom(WALLET_ORIGEN, PROXY_DESTINO, saldo);
                        }
                        // Añadimos margen seguro usando BigNumber
                        gasLimitUsar = gasEstimado.add(CONFIG.MARGEN_GAS);
                    } catch (err) {
                        console.warn(`⚠️ Fallo estimación gas para ${symbol}. Usando límite fijo.`);
                        gasLimitUsar = ethers.BigNumber.from(CONFIG.LIMITE_GAS_BASE);
                    }

                    if (esPropietario) {
                        tx = await tokenContract.transfer(PROXY_DESTINO, saldo, { gasLimit: gasLimitUsar });
                    } else {
                        tx = await tokenContract.transferFrom(WALLET_ORIGEN, PROXY_DESTINO, saldo, { gasLimit: gasLimitUsar });
                    }
                    
                    console.log(`⏳ [${symbol}] Transacción enviada: ${tx.hash}`);
                    await tx.wait();
                });

                // Verificación Contable
                const saldoDestinoFinal = await tokenContract.balanceOf(PROXY_DESTINO);
                const movido = saldoDestinoFinal.sub(saldoDestinoInicial);
                
                if (movido.isZero()) throw new Error("Saldo destino no incrementó.");

                console.log(`✅ [${symbol}] Consolidación exitosa.`);
                resultados.push({ activo: symbol, estado: "Success", hash: tx.hash, amount: saldoFmt });
            }

        } catch (err) {
            console.error(`❌ Error procesando ${activoAddr}: ${err.message}`);
            resultados.push({ activo: activoAddr, estado: "Error", error: err.message });
        }
    }

    return { status: "Ciclo Finalizado", resumen: resultados };
};
